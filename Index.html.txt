<script>

let runs = 0;
let wickets = 0;
let balls = 0;

function updateScore() {
    document.getElementById("score").innerHTML = runs + "/" + wickets;

    let over = Math.floor(balls / 6);
    let ball = balls % 6;

    document.getElementById("overs").innerHTML = over + "." + ball;
}

function addRun(run) {
    runs += run;
    balls++;
    updateScore();
}

function dotBall() {
    balls++;
    updateScore();
}

function wideBall() {
    runs += 1;
    updateScore();
}

function noBall() {
    runs += 1;
    updateScore();
}

function wicket() {
    wickets++;
    balls++;
    updateScore();
}

function resetMatch() {
    runs = 0;
    wickets = 0;
    balls = 0;
    updateScore();
}

</script>
